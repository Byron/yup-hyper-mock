An implementation of Ssl for OpenSSL.

Example

use hyper::Server;
use hyper::net::Openssl;

let ssl = Openssl::with_cert_and_key("/home/foo/cert", "/home/foo/key").unwrap();
Server::https("0.0.0.0:443", ssl).unwrap();
