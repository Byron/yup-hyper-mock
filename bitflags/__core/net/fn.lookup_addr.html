pub fn lookup_addr(addr: &IpAddr) -> Result<String, Error>
Deprecated since 1.6.0

: ipaddr type is being deprecated

Resolve the given address to a hostname.

This function may perform a DNS query to resolve addr and may also inspect system configuration to resolve the specified address. If the address cannot be resolved, it is returned in string format.

Examples

#![feature(lookup_addr)]
#![feature(ip_addr)]

use std::net::{self, Ipv4Addr, IpAddr};

let ip_addr = "8.8.8.8";
let addr: Ipv4Addr = ip_addr.parse().unwrap();
let hostname = net::lookup_addr(&IpAddr::V4(addr)).unwrap();

println!("{} --> {}", ip_addr, hostname);
// Output: 8.8.8.8 --> google-public-dns-a.google.com
